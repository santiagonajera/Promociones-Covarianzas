import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
from io import BytesIO
import requests

# URL del archivo Excel
url = "https://github.com/santiagonajera/Promociones-Covarianzas/raw/refs/heads/main/datos_temperaturas.xlsx"

# Descargar el archivo
response = requests.get(url)
excel_data = BytesIO(response.content)

# Leer el archivo Excel
datos = pd.read_excel(excel_data)

# Convertir la columna 'fecha' a tipo datetime si no lo es ya
datos['fecha'] = pd.to_datetime(datos['fecha'])

# Calcular matriz de covarianza
columnas_numericas = ['temperatura', 'agua_normal', 'agua_sabor', 'eventos_especiales']
matriz_cov = datos[columnas_numericas].cov()

# Visualizar matriz de covarianza
plt.figure(figsize=(10, 8))
sns.heatmap(matriz_cov, annot=True, cmap='coolwarm', fmt='.2f')
plt.title('Matriz de Covarianza en la Cadena de Suministro')
plt.show()

# Calcular correlacin entre productos
correlacion = datos['agua_normal'].corr(datos['agua_sabor'])
print(f"Correlacin entre Agua Normal y Agua con Sabor: {correlacion:.2f}")

# Analizar el impacto de la temperatura en la demanda
plt.figure(figsize=(12, 6))
plt.scatter(datos['temperatura'], datos['agua_normal'], alpha=0.5, label='Agua Normal')
plt.scatter(datos['temperatura'], datos['agua_sabor'], alpha=0.5, label='Agua con Sabor')
plt.xlabel('Temperatura')
plt.ylabel('Demanda')
plt.title('Impacto de la Temperatura en la Demanda')
plt.legend()
plt.show()

# Calcular el impacto de eventos especiales
impacto_normal = datos[datos['eventos_especiales'] == 1]['agua_normal'].mean() - datos[datos['eventos_especiales'] == 0]['agua_normal'].mean()
impacto_sabor = datos[datos['eventos_especiales'] == 1]['agua_sabor'].mean() - datos[datos['eventos_especiales'] == 0]['agua_sabor'].mean()
print(f"Impacto promedio de eventos especiales en Agua Normal: {impacto_normal:.2f}")
print(f"Impacto promedio de eventos especiales en Agua con Sabor: {impacto_sabor:.2f}")

# Clculo de una matriz de covarianza "robusta" simplificada
def covarianza_robusta(X):
    medias = np.median(X, axis=0)
    X_centrada = X - medias
    return np.cov(X_centrada.T)

cov_robusta = covarianza_robusta(datos[columnas_numericas])

print("\nMatriz de Covarianza Robusta Simplificada:")
print(cov_robusta)

# Calcular y visualizar la demanda conjunta
datos['demanda_total'] = datos['agua_normal'] + datos['agua_sabor']

plt.figure(figsize=(12, 6))
plt.plot(datos['fecha'], datos['demanda_total'])
plt.title('Demanda Total a lo Largo del Tiempo')
plt.xlabel('Fecha')
plt.ylabel('Demanda Total')
plt.show()

# Calcular la covarianza mvil
tamanio_ventana = 30
cov_movil = datos[['agua_normal', 'agua_sabor']].rolling(window=tamanio_ventana).cov().unstack()['agua_normal']['agua_sabor']

plt.figure(figsize=(12, 6))
plt.plot(datos['fecha'][tamanio_ventana-1:], cov_movil[tamanio_ventana-1:])
plt.title(f'Covarianza Mvil entre Agua Normal y Agua con Sabor (Ventana de {tamanio_ventana} das)')
plt.xlabel('Fecha')
plt.ylabel('Covarianza')
plt.show()

# Imprimir insights y recomendaciones
print("\nInsights y Recomendaciones:")
print("1. Planificacin de Inventario:")
if correlacion > 0.5:
    print("   - Alta correlacin entre Agua Normal y Agua con Sabor. Considere mantener niveles de inventario similares.")
else:
    print("   - Baja correlacin entre Agua Normal y Agua con Sabor. Puede ser necesario gestionar inventarios por separado.")

print("2. Impacto de la Temperatura:")
if matriz_cov.loc['temperatura', 'agua_normal'] > 0:
    print("   - La demanda de Agua Normal tiende a aumentar con la temperatura. Prepare ms inventario en temporadas clidas.")
if matriz_cov.loc['temperatura', 'agua_sabor'] < 0:
    print("   - La demanda de Agua con Sabor tiende a disminuir con la temperatura. Reduzca inventario en temporadas clidas.")

print("3. Eventos Especiales:")
if impacto_normal > 0 or impacto_sabor > 0:
    print("   - Los eventos especiales tienen un impacto significativo en la demanda. Planifique inventario adicional para estos das.")

print("4. Anlisis de Tendencias:")
print("   - Utilice la covarianza mvil para identificar cambios en las relaciones entre productos a lo largo del tiempo.")

print("5. Gestin de Riesgos:")
print("   - La matriz de covarianza robusta simplificada puede ayudar a identificar relaciones ms estables entre variables, til para la planificacin a largo plazo.")
